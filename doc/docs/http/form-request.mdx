# Form Request

Pada saat melakukan request sering kali membutuhkan validasi saat request berhasil atau gagal. Berikut adalah contoh penggunaan form request:

import Tabs from "@theme/Tabs";
import TabItem from "@theme/TabItem";

<Tabs>
  <TabItem value="html" label="Html" default>
    ```html
    <form id="form">
      <input type="text" name="name" placeholder="Name" validate="name">
      <input type="email" name="email" placeholder="Email" validate="email">
      <button type="submit">Submit</button>
    </form>
    ```
  </TabItem>
  <TabItem value="javascript" label="JS">
    ```js
      $("#form").on('submit',function(e){
        e.preventDefault();
        iam.request({
          method: "POST",
          url: "https://jsonplaceholder.typicode.com/posts",
          data: new FormData(this),
          onSuccess: function (res) {
            // Do something
          },
          onError: function (res) {
            // Do something
            iam.utils.validate(res.responseJSON); // Validation on error
          },
        })
        
      });
    ```
  </TabItem>
</Tabs>
